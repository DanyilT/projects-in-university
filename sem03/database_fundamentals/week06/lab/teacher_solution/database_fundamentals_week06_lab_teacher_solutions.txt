SELECT empno, ename
FROM emp
WHERE ename LIKE'b%' AND job NOT IN('clerk','president','manager');

#Query 2
SELECT ename, sal, (sal *12) AS 'Annual Salary'
FROM emp;

#Query 3
SELECT ename, job, deptno, hiredate
FROM emp
ORDER BY ename;

#Query 4

#QUERY 5
SELECT ename, FLOOR((DATEDIFF(curdate(),hiredate)/365)) AS 'Years Employed'
FROM emp;

#Query 6

#Query 7

#Query 8
